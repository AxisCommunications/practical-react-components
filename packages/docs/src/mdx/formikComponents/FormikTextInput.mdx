# FormikTextInput

A formik component that wraps an TextInput component.

## Demo

```typescript type=demo
<FormikDemo initialValues={{ name: '' }}>
  <FormikTextInputField name="name" label="Nickname" />
</FormikDemo>
```

## Usage in code

```typescript
import React, { useMemo, useCallback } from 'react'
import { useFormik, FormikProvider } from 'formik'
import { FormikTextInputField } from 'practical-react-components-formik'
import * as yup from 'yup'

const MyComponent = ({ myinput }) => {
  const validationSchema = useMemo(
    () =>
      yup.object().shape({
        myinput: yup.string().required(),
      }),
    []
  )

  const initialValues = useMemo(
    () => ({
      myinput,
    }),
    // Initial values that should not change during
    // re-renders as formik handles state by itself
    // eslint-disable-next-line react-hooks/exhaustive-deps
    []
  )

  const onSubmit = useCallback((values: IFormikValues) => {
    /* Save data here */
  }, [])

  const formik = useFormik({
    initialValues,
    validationSchema,
    onSubmit,
  })

  return (
    <FormikProvider value={formik}>
      <FormikTextInputField label="Input" name="myinput" />
    </FormikProvider>
  )
}
```
