# FormikDurationInput

A formik component that uses DurationInputField component.
The time input uses a text input box to write time, the input forces the
user to enter in a specific format e.g. (hh:mm:ss).

## Demo

```typescript type=demo
<FormikDemo initialValues={{ duration: 66 }}>
  <FormikDurationInputField
    name="duration"
    format="hh:mm:ss"
    label="Duration between retries"
    unitLabel="HH:MM:SS"
  />
</FormikDemo>
```

## Usage in code

```typescript
import { useMemo, useCallback } from 'react'
import { useFormik, FormikProvider } from 'formik'
import { FormikDurationInputField } from 'practical-react-components-formik'
import * as yup from 'yup'

const MyComponent = () => {
  const validationSchema = useMemo(
    () =>
      yup.object().shape({
        seconds: yup.number().required(),
      }),
    []
  )

  const initialValues = useMemo(
    () => ({
      seconds: 1,
    }),
    // Initial values that should not change during
    // re-renders as formik handles state by itself
    // eslint-disable-next-line react-hooks/exhaustive-deps
    []
  )

  const onSubmit = useCallback((values: IFormikValues) => {
    /* Save data here */
  }, [])

  const formik = useFormik({
    initialValues,
    validationSchema,
    onSubmit,
  })

  return (
    <FormikProvider value={formik}>
      <FormikDurationInputField
        name="seconds"
        format="hh:mm"
        label="Time"
        unitLabel="HH:MM"
      />
    </FormikProvider>
  )
}
```
