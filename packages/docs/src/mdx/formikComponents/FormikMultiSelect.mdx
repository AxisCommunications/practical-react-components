# FormikMultiSelect

A formik component that wraps a MultiSelect component.

## Demo

```typescript type=demo
<FormikDemo initialValues={{ fruits: [] }}>
  <FormikMultiSelectField
    label="Select favorite fruits"
    name="fruits"
    options={[
      { label: 'Apple', value: 'apple' },
      { label: 'Banana', value: 'banana' },
      { label: 'Cherry', value: 'cherry' },
      { label: 'Dragon fruit', value: 'dragon_fruit' },
      { label: 'Elderberry', value: 'elderberry' },
    ]}
  />
</FormikDemo>
```

## Usage in code

```typescript
import React, { useMemo, useCallback } from 'react'
import { useFormik, FormikProvider } from 'formik'
import { FormikMultiSelectField } from 'practical-react-components-formik'
import * as yup from 'yup'

const MyComponent = ({ myvalues }) => {
  const validationSchema = useMemo(
    () =>
      yup.object().shape({
        myvalues: yup.array().of(yup.string()),
      }),
    []
  )

  const initialValues = useMemo(
    () => ({
      myvalues,
    }),
    // Initial values that should not change during
    // re-renders as formik handles state by itself
    // eslint-disable-next-line react-hooks/exhaustive-deps
    []
  )

  const onSubmit = useCallback((values: FormikValues) => {
    /* Save data here */
  }, [])

  const formik = useFormik({
    initialValues,
    validationSchema,
    onSubmit,
  })

  const options = [
    { label: 'Apple', value: 'apple' },
    { label: 'Banana', value: 'banana' },
    { label: 'Cherry', value: 'cherry' },
    { label: 'Dragon fruit', value: 'dragon_fruit' },
    { label: 'Elderberry', value: 'elderberry' },
  ]

  return (
    <FormikProvider value={formik}>
      <FormikMultiSelectField
        label="MultiSelect"
        name="myvalues"
        options={options}
      />
    </FormikProvider>
  )
}
```
