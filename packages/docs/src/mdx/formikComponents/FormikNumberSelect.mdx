# FormikNumberSelect

A formik component that wraps a Select component.
The difference from a normal Select and this component is that it takes option values as numbers.

## Demo

```typescript type=demo
<FormikDemo initialValues={{ day: 1 }}>
  <FormikNumberSelectField
    label="Day"
    name="day"
    options={[
      { label: 'Monday', value: 1 },
      { label: 'Tuesday', value: 2 },
      { label: 'Wednesday', value: 3 },
      { label: 'Thursday', value: 4 },
      { label: 'Friday', value: 5 },
    ]}
  />
</FormikDemo>
```

## Usage in code

```typescript
import React, { useMemo, useCallback } from 'react'
import { useFormik, FormikProvider } from 'formik'
import { FormikNumberSelectField } from 'practical-react-components-formik'
import * as yup from 'yup'

const MyComponent = ({ enabled }) => {
  const validationSchema = useMemo(
    () =>
      yup.object().shape({
        selected: yup.string().required(),
      }),
    []
  )
  const options = useMemo(
    () => [
      { label: 'January', value: 1 },
      { label: 'February', value: 2 },
      { label: 'March', value: 3 },
      { label: 'April', value: 4 },
      { label: 'May', value: 5 },
      { label: 'June', value: 6 },
      { label: 'July', value: 7 },
      { label: 'August', value: 8 },
      { label: 'September', value: 9 },
      { label: 'October', value: 10 },
      { label: 'November', value: 11 },
      { label: 'December', value: 12 },
    ],
    []
  )

  const initialValues = useMemo(
    () => ({
      month: 1,
    }),
    // Initial values that should not change during
    // re-renders as formik handles state by itself
    // eslint-disable-next-line react-hooks/exhaustive-deps
    []
  )

  const onSubmit = useCallback((values: IFormikValues) => {
    /* Save data here */
  }, [])

  const formik = useFormik({
    initialValues,
    validationSchema,
    onSubmit,
  })

  return (
    <FormikProvider value={formik}>
      <FormikNumberSelectField
        label="Component label"
        name="month"
        options={options}
      />
    </FormikProvider>
  )
}
```
