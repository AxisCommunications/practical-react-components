# FormikInputCredentials

A formik component that wraps a InputCredentials component.

## Demo

```typescript type=demo
<FormikDemo initialValues={{ password: 'secret' }}>
  <FormikInputCredentialsField
    label="Password"
    name="password"
    type="new-password"
    placeholder="Placeholder"
  />
</FormikDemo>
```

## Usage in code

```typescript
import React, { useMemo, useCallback } from 'react'
import { useFormik, FormikProvider } from 'formik'
import { FormikInputCredentialsField } from 'practical-react-components-formik'
import * as yup from 'yup'

const MyComponent = ({ password }) => {
  const validationSchema = useMemo(
    () =>
      yup.object().shape({
        password: yup.string().required(),
      }),
    []
  )

  const initialValues = useMemo(
    () => ({
      password,
    }),
    // Initial values that should not change during
    // re-renders as formik handles state by itself
    // eslint-disable-next-line react-hooks/exhaustive-deps
    []
  )

  const onSubmit = useCallback((values: FormikValues) => {
    /* Save data here */
  }, [])

  const formik = useFormik({
    initialValues,
    validationSchema,
    onSubmit,
  })

  return (
    <FormikProvider value={formik}>
      <FormikInputCredentialsField
        label="Password"
        name="password"
        type="new-password"
        placeholder="Placeholder"
      />
    </FormikProvider>
  )
}
```
