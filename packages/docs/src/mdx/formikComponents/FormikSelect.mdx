# FormikSelect

A formik component that wraps a Select component.

## Demo

```typescript type=demo
<FormikDemo initialValues={{ fruits: 'cherry' }}>
  <FormikSelectField
    label="Select favorite fruits"
    name="fruits"
    options={[
      { label: 'Apple', value: 'apple' },
      { label: 'Banana', value: 'banana' },
      { label: 'Cherry', value: 'cherry' },
      { label: 'Dragon fruit', value: 'dragon_fruit' },
      { label: 'Elderberry', value: 'elderberry' },
    ]}
  />
</FormikDemo>
```

## Usage in code

```typescript
import React, { useMemo, useCallback } from 'react'
import { useFormik, FormikProvider } from 'formik'
import { FormikSelectField } from 'practical-react-components-formik'
import * as yup from 'yup'

const MyComponent = ({ enabled }) => {
  const validationSchema = useMemo(
    () =>
      yup.object().shape({
        selected: yup.string().required(),
      }),
    []
  )

  const initialValues = useMemo(
    () => ({
      selected,
    }),
    // Initial values that should not change during
    // re-renders as formik handles state by itself
    // eslint-disable-next-line react-hooks/exhaustive-deps
    []
  )

  const onSubmit = useCallback((values: FormikValues) => {
    /* Save data here */
  }, [])

  const formik = useFormik({
    initialValues,
    validationSchema,
    onSubmit,
  })

  const options = [
    { label: 'Apple', value: 'apple' },
    { label: 'Banana', value: 'banana' },
    { label: 'Cherry', value: 'cherry' },
    { label: 'Dragon fruit', value: 'dragon_fruit' },
    { label: 'Elderberry', value: 'elderberry' },
  ]

  return (
    <FormikProvider value={formik}>
      <FormikSelectField label="Selected" name="selected" options={options} />
    </FormikProvider>
  )
}
```
