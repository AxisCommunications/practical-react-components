# Modal

The modal component provides a solid foundation for creating dialogs, popovers, lightboxes, or whatever else.

export const DemoComponent = ({}) => {
  const [open, setOpen, unsetOpen] = useBoolean(false)
  const [openSub, unsetOpenSub] = useBoolean(false)

}

## Usage example

```typescript
import { Modal, Button } from 'practical-react-components-core'

export const DemoComponent = () => {
  const [showModal, setShowModal] = useState(false)
  const setModal = () => setShowModal(!showModal)
  return (
    <>
      <Button label="Open modal" onClick={setModal} variant="primary" />
      <Modal
        open={showModal}
        onClose={setModal}
        title="modal-level-1"
        focusDialog={true}
      >
        CONTENT
        <Button label="Close modal" onClick={setModal} variant="primary" />
      </Modal>
    </>
  )
}
```
