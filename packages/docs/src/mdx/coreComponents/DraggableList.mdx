# DraggableList

A list that allows re-ordering of its items.

## Demo

export const DraggableDemo = () => {
  const [state, setState] = useState([0, 1, 2, 3, 4])
  return (
    <DraggableList
      onChange={nextIndices => {
        setState(nextIndices.map(index => state[index]))
      }}
    >
      {state.map((item, index) => (
        <DraggableListItem key={index} disabled={item === 3}>
          <Typography>{item}</Typography>
        </DraggableListItem>
      ))}
    </DraggableList>
  )
}

<DraggableDemo />
