# Tab

Tabs make it easy to explore and switch between different views.

Tabs organize and allow navigation between groups of content that are
related and at the same level of hierarchy.

## Demo

export const HorizontalTabContainer = styled.div`
  display: flex;
  > * {
    width: 128px;
    margin: 8px;
  }
`

export const VerticalTabContainer = styled.div`
  display: inline-flex;
  flex-direction: column;
  > * {
    width: 128px;
    margin: 8px;
  }
`

export const VContainer = styled.div`
  display: flex;
  align-items: center;
`

export const HTABS = [
  {
    id: '1',
    label: 'Home',
    icon: HomeIcon,
  },
  {
    id: '2',
    label: 'Contact',
    icon: SystemAccessoriesIcon,
  },
]

export const VTABS = [
  {
    id: '1',
    label: 'Home',
    icon: HomeIcon,
  },
  {
    id: '2',
    label: 'Contact',
    icon: SystemAccessoriesIcon,
  },
]

export const Container = styled.div`
  display: ${({ show }) => (show ? 'unset' : 'none')};
`

export const ExamplePaper = styled(Paper)`
  background-color: ${({ theme }) => theme.color.background03()};
`

export const TabPaper = styled(Paper)`
  max-width: 500px;
  margin: 0 12px;
`

export const DemoComponent = ({}) => {
  const [selectedHTab, setSelectedHTab] = useState('1')
  const [selectedVTab, setSelectedVTab] = useState('1')
  return (
    <>
      <ExamplePaper space={true}>
        <HorizontalTabContainer>
          {HTABS.map(tab => (
            <Tab
              key={tab.id}
              id={tab.id}
              label={tab.label}
              icon={tab.icon}
              selected={tab.id === selectedHTab}
              onSelect={setSelectedHTab}
              className="my-tab"
            />
          ))}
        </HorizontalTabContainer>
        <Container show={selectedHTab === '1'}>
          <TabPaper space={true}>
            <Typography variant="page-heading">Home tab</Typography>
            <Typography>This is the content of the Home tab.</Typography>
          </TabPaper>
        </Container>
        <Container show={selectedHTab === '2'}>
          <TabPaper space={true}>
            <Typography variant="page-heading">Contact tab</Typography>
            <Typography>This is the content of the Contact tab.</Typography>
          </TabPaper>
        </Container>
      </ExamplePaper>
      <br />
      <ExamplePaper space={true}>
        <VContainer>
          <VerticalTabContainer>
            {VTABS.map(tab => (
              <VerticalTab
                key={tab.id}
                id={tab.id}
                label={tab.label}
                icon={tab.icon}
                selected={tab.id === selectedVTab}
                onSelect={setSelectedVTab}
              />
            ))}
          </VerticalTabContainer>
          <Container show={selectedVTab === '1'}>
            <TabPaper space={true}>
              <Typography variant="page-heading">Home tab</Typography>
              <Typography>This is the content of the Home tab.</Typography>
            </TabPaper>
          </Container>
          <Container show={selectedVTab === '2'}>
            <TabPaper space={true}>
              <Typography variant="page-heading">Contact tab</Typography>
              <Typography>This is the content of the Contact tab.</Typography>
            </TabPaper>
          </Container>
        </VContainer>
      </ExamplePaper>
    </>
  )
}

<DemoComponent />

## Basic usage

```typescript type=live
<Tab
  id="id1"
  label="Home"
  icon={HomeIcon}
  selected={true}
  onSelect={() => {}}
/>
```
